go_push 一个实用的消息推送服务

go_push (推送服务)

    ├── gateway // 长连接网关服务器
    │   ├── push_job.go    // 分发任务
    │   ├── room.go        // 房间，可作为某一类型的推送管理中心
    │   ├── room_manage.go // 房间管理
    │   ├── ws_conn.go     // 简单封装的websocket方法
    │   ├── ws_handle.go   // 处理websocket协议方法
    │   └── ws_server.go   // websocket服务
    ├── logic  //逻辑服务器    
    │   ├── http_handle.go // 推送，房间相关
    │   └── http_server.go // http服务
    └── main.go
